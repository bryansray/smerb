  def server_list refresh = false
    @server_list = nil if refresh
    @server_list ||= XMLRPC::Client.new("xmlrpc.mollom.com", "/#{API_VERSION}").call('mollom.getServerList', authentication_hash).collect do |server| 
      proto, ip = server.split('://')
      {:proto => proto, :ip => ip}
    end
  end
